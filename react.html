We can create our UI custom components then reuse them. Seems that we create html tags that have our logic inside.

function LocalTime() {
  return (
    <div>
      It is {new Date().toLocaleTimeString()}.
    </div>
  );
}

// then use it as a simple html tag
const element = (
  <div>
    <LocalTime />
  </div>
);

Components and Props

function components

function Welcome(props) {
  return <h1>Hello, {props.name}</h1>;
}
// or using arrow fn (implicit return & destructoring)
const Welcome = ({name}) => (
  <h1>Hello, {name}</h1>
)

ES6 class components

class Welcome extends React.Component {
  render() {
    return <h1>Hello, {this.props.name}</h1>;
  }
}

const element = <Welcome name="David" />; // usage for any of them

ReactDOM.render(
  element,
  document.getElementById('root')
