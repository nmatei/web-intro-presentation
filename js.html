webpack.config.js

                const path = require("path");
                const HtmlWebpackPlugin = require("html-webpack-plugin");

                module.exports = env => {
                  const isProduction = !!env.WEBPACK_BUILD;
                  return {
                    mode: isProduction ? "production" : "development",
                    entry: ["./src/index.js"],
                    devtool: isProduction ? false : "inline-source-map",
                    devServer: {
                      static: ["dist", "src"],
                      watchFiles: ["src/**/*.*"]
                    },
                    plugins: [
                      new HtmlWebpackPlugin({
                        template: "./src/index.html"
                      })
                    ],
                    output: {
                      filename: "main.js",
                      path: path.resolve(__dirname, "dist"),
                      publicPath: "/"
                    }
                  };
                };

Fetch and Promises

Fetch response ( GET / POST / PUT / DELETE ) are Promises


                // step by step request using fetch
                const request = fetch("http://localhost:3000/teams-json", {
                  method: "GET",
                  headers: {
                    "Content-Type": "application/json"
                  }
                });
                console.info(typeof request, request);
                // object, Promise{<pending>}

                const response = request.then(function (response) {
                  return response.json();
                });
                console.info(typeof response, response);

                const promise = response.then(function(teams) {
                    console.info('teams', teams);
                });
                console.info(typeof promise, promise);
