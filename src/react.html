    React.Component & TypeScript

    type Props = {
    name: string;
}
type State = {
    loading: boolean
}
class Welcome extends React.Component <Props, State> {
  constructor(props) {
    super(props);
    this.state = {
      loading: true
    }
  }
  render() {
    const cls = this.state.loading ? "loading-mask" : "";
    return <h1 className={cls}>Hello, {this.props.name}</h1>;
  }
}
// usage
<Welcome name="David" />

    Multiple values

    export type Team = {
  id?: string;
  members: string;
  name: string;
  amount?: number;
};

type TeamKeys = "id" | "members" | "name" | "amount";

function getValue(team: Team, property: TeamKeys) {
  return team[property];
}

// improve...
function setValue(team: Team, property: keyof Team, value: any) {
  team[property as string] = value;
}

    *keyof

    function updateValue<T, K extends keyof T>(obj: T, key: K, value: T[K]) {
  obj[key] = value;
}

const point = {
  x: 1,
  y: 1
};
updateValue(point, "x", 2); // try other type for values

const team: Team = {
  members: "gr6",
  name: "ts"
};

updateValue(team, "members", "Str");
updateValue(team, "amount", 123);

updateValue(team, "amount", "456");
//  Argument of type 'string' is not assignable to parameter of type 'number'

    more about Generics + Partial

    type Store<T = any> = {
  find<K extends keyof T>(fieldName: K, value: T[K]): T;
  update(id: string, values: Partial<T>): void;
};

// const store = getStore(); // ... new CustomStore()
const store: Store<Team> = {
  find(fieldName, values) {
    return {
      name:"y",
      members:"we"
    };
  },
  update(id, value) {}
};

store.find("id", "12345");
store.find("amount", 20220);
store.find("not-found", "value");
// Argument of type '"not-found"' is not assignable to...
store.update("123", { amount: "1" });
// Type 'string' is not assignable to type 'number'
