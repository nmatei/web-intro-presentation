    debounce (imports & exports)

    export function debounce(fn, delay) {
  var timer = null; // 2Ô∏è‚É£ Closures
  return function () {
    // 3Ô∏è‚É£ context (this)
    var context = this,
      args = arguments;
    clearTimeout(timer);
    timer = setTimeout(function () {
      fn.apply(context, args); // 1Ô∏è‚É£ Callback function
    }, delay);
  };
}
